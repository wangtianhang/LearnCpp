#version 450 core
// Declare VS_OUT as an input interface block
in VS_OUT
{
vec4 color; // Send color to the next stage
vec2 uv;
} fs_in;
uniform sampler2D texture1;
uniform ivec2 texSize2;
// Output to the framebuffer
out vec4 color;

void main(void)
{
// Simply assign the color we were given by the vertex shader to our output
	vec3 result = vec3(0.0);
    for(int x=-1; x<=1; x++)
	{
        for(int y=-1; y<=1; y++)
		{
            vec2 off = vec2(x,y)/texSize2;
			result += texture(texture1,fs_in.uv + off).rgb;
		}
	}
	vec3 tmp = result / 9;
	color = vec4(tmp, 1);
}
